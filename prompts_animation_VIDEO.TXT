Esta guía documenta cómo construir una página web donde un video se reproduce frame por frame sincronizado con el scroll del usuario — idéntico a las páginas de producto de Apple. El resultado final es un sitio completo con secciones de hero, animación de scroll, features y CTA, totalmente personalizable.

El resultado final incluye:
- Página web con fondo de color a elección y branding propio
- Sección de 600vh donde el canvas queda fijo (sticky) y el video avanza frame por frame al scrollear
- Video renderizado en modo cover — ocupa toda la pantalla sin deformación ni barras negras
- Pantalla de carga con barra de progreso mientras precargan todos los frames
- Resolución retina/HiDPI — nítido en pantallas de alta densidad de píxeles
- Secciones: Navbar, Hero, Kicker, Animación, Features, CTA, Footer

Arquitectura general (3 partes):
- Extracción de frames: ffmpeg — extrae el video como imágenes individuales numeradas
- Servidor local: python3 -m http.server — sirve los archivos via HTTP local
- Animación web: HTML + CSS + JS puro — precarga frames, detecta scroll, dibuja el frame correcto en canvas

Estructura de archivos:
mi-proyecto/
├── index.html
└── frames/
    ├── frame_0001.jpg
    └── ... (hasta frame_NNNN.jpg)

PASO 1: Analizar el video con ffprobe
ffprobe -v quiet -print_format json -show_streams "ruta/al/video.mp4"
Anotar: nb_frames (total de frames), width x height (resolución nativa), r_frame_rate (fps).

PASO 2: Extraer frames con ffmpeg
mkdir frames
ffmpeg -i "ruta/al/video.mp4" \
  -an \
  -vf "fps=FPS,scale=ANCHO:ALTO" \
  -c:v mjpeg \
  -q:v 2 \
  "frames/frame_%04d.jpg"
Regla: escalar a la mitad de la resolución nativa (sigue siendo perfecta en retina).

PASO 3: Servidor local — OBLIGATORIO
cd ruta/a/mi-proyecto/
python3 -m http.server 8765
Nunca abrir index.html con doble click (file://). Chrome bloquea la carga de imágenes locales.

PASO 5: JavaScript completo de la animación
Solo cambian 4 valores: TOTAL_FRAMES, NATIVE_W, NATIVE_H, FRAMES_DIR

(() => {
const TOTAL_FRAMES = 192;
const NATIVE_W = 1080;
const NATIVE_H = 1920;
const FRAMES_DIR = './frames/';
const pad = n => String(n).padStart(4, '0');

const canvas = document.getElementById('frame-canvas');
const ctx = canvas.getContext('2d', { alpha: false });
const loader = document.getElementById('loader');
const fill = document.getElementById('loader-fill');
const pct = document.getElementById('loader-pct');
const hint = document.getElementById('scroll-hint');
const section = document.getElementById('scroll-section');

let dpr, vpW, vpH;
function sizeCanvas() {
    dpr = window.devicePixelRatio || 1;
    vpW = window.innerWidth;
    vpH = window.innerHeight;
    canvas.style.width = vpW + 'px';
    canvas.style.height = vpH + 'px';
    canvas.width = Math.round(vpW * dpr);
    canvas.height = Math.round(vpH * dpr);
    ctx.scale(dpr, dpr);
}

function drawCover(img) {
    if (!img || !img.naturalWidth) return;
    const imgAspect = NATIVE_W / NATIVE_H;
    const canvasAspect = vpW / vpH;
    let srcX, srcY, srcW, srcH;
    if (canvasAspect > imgAspect) {
        srcW = NATIVE_W; srcH = NATIVE_W / canvasAspect;
        srcX = 0; srcY = (NATIVE_H - srcH) / 2;
    } else {
        srcH = NATIVE_H; srcW = NATIVE_H * canvasAspect;
        srcY = 0; srcX = (NATIVE_W - srcW) / 2;
    }
    ctx.drawImage(img, srcX, srcY, srcW, srcH, 0, 0, vpW, vpH);
}

let sectionTop = 0, scrollRange = 0;
function updateMetrics() {
    sectionTop = section.offsetTop;
    scrollRange = section.offsetHeight - window.innerHeight;
}

const frames = new Array(TOTAL_FRAMES);
let loadedCount = 0;
function preload() {
    return new Promise(resolve => {
        for (let i = 0; i < TOTAL_FRAMES; i++) {
            const img = new Image();
            img.onload = img.onerror = () => {
                loadedCount++;
                const p = Math.round((loadedCount / TOTAL_FRAMES) * 100);
                fill.style.width = p + '%';
                pct.textContent = p + '%';
                if (loadedCount === TOTAL_FRAMES) resolve();
            };
            img.src = FRAMES_DIR + 'frame_' + pad(i + 1) + '.jpg';
            frames[i] = img;
        }
    });
}

let currentIdx = -1;
function draw(idx) {
    if (idx === currentIdx) return;
    currentIdx = idx;
    drawCover(frames[idx]);
}

let rafPending = false, nextIdx = 0;
function onScroll() {
    const scrolled = window.scrollY - sectionTop;
    const progress = Math.max(0, Math.min(1, scrolled / scrollRange));
    nextIdx = Math.min(TOTAL_FRAMES - 1, Math.floor(progress * TOTAL_FRAMES));
    if (scrolled > 20) hint.classList.add('gone');
    if (!rafPending) {
        rafPending = true;
        requestAnimationFrame(() => { draw(nextIdx); rafPending = false; });
    }
}

window.addEventListener('resize', () => {
    sizeCanvas(); updateMetrics();
    const saved = currentIdx < 0 ? 0 : currentIdx;
    currentIdx = -1; draw(saved);
});

sizeCanvas();
preload().then(() => {
    updateMetrics(); currentIdx = -1; draw(0);
    loader.classList.add('hidden');
    onScroll();
    window.addEventListener('scroll', onScroll, { passive: true });
});
})();

ERRORES CRÍTICOS A EVITAR:
1. No abrir con file:// -> usar python3 -m http.server
2. No usar getBoundingClientRect() para scroll -> usar window.scrollY - section.offsetTop
3. No verificar con img.complete -> verificar con img.naturalWidth
4. ctx.scale() se acumula en resize -> asignar canvas.width resetea el contexto
5. No extraer a resolución 4K nativa -> usar mitad de resolución
6. Flag "ticking" bloqueado -> usar rafPending con reset garantizado
